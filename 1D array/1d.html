                #include <stdio.h>

                    int main() {
                        // ---------------- Integer array section ----------------
                        int n;
                    
                        // Get the size of the array
                        printf("Enter the number of elements: ");
                        scanf("%d", &n);
                    
                        int intArr[n];  // Renamed integer array
                    
                        // Take integer array input
                        printf("Enter %d integers:\n", n);
                        for (int i = 0; i < n; i++) {
                            scanf("%d", &intArr[i]);
                        }
                    
                        // Print integers horizontally
                        printf("\nInteger array in horizontal form:\n");
                        for (int i = 0; i < n; i++) {
                            printf("%d ", intArr[i]);
                        }
                    
                        // Print integers vertically
                        printf("\n\nInteger array in vertical form:\n");
                        for (int i = 0; i < n; i++) {
                            printf("%d\n", intArr[i]);
                        }
                    
                        // ---------------- Character array section ----------------
                        char charArr[100];  // Renamed character array
                    
                        // Take string input
                        printf("\nEnter a string: ");
                        scanf("%s", charArr);  // Takes input without spaces
                    
                        // Print string horizontally
                        printf("\nString in horizontal form:\n");
                        printf("%s\n", charArr);
                    
                        // Print string vertically
                        printf("\nString in vertical form:\n");
                        for (int i = 0; charArr[i] != '\0'; i++) {
                            printf("%c\n", charArr[i]);
                        }
                    
                        return 0;
                    }
